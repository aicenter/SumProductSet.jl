function train!(m, x; niter::Int=100, opt=ADAM(1.))
    ps = Flux.params(m)

    for i in 1:niter
        gs = gradient(() -> SumProductSet.ul_loss(m, x), ps)
        Flux.Optimise.update!(opt, ps, gs)
    end
end

predict = x->mapslices(argmax, logjnt(m, x), dims=1)[:]
